import { Component } from '@angular/core';
import { CommonModule } from '@angular/common';

interface Product {
  id: number;
  name: string;
  image: string;
  details: string;
}

@component({
  selector: 'app-root',
  standalone: true,
  imports: [CommonModule],
  template:
  `<div class='flex h-screen'>
        <!-- Product List -->
        <div class='flex-1 border-l border-gray-300 p-4 overflow-y-auto'>
          <ul class='list-none p-0'>
            <li
              *ngFor='let product of products'
              (click)='selectProduct(product)'
              class='cursor-pointer p-2 border-b border-gray-300 flex items-center hover:bg-gray-100'
            >
            <img
              [src]='product.image'
              alt='{{ product.name }}'
              class='w-12 h-12 object-cover mr-4 rounded'
            />
              <span class='text-lg font-medium'>{{ 'product.name' }}</span>
            </li>
          </ul>
        </div>
        <div class='flex-2 border-l border-gray-300 p-4'>
          <ng-container *ngIf='selectedProduct; else noSelection'>
            <h2 class='text-2xl font-semibold mb-4'>{{ 'selectedProduct.name' }}</h2>
            <img
              [src]='selectedProduct.image'
              alt='{{ selectedProduct.name }}'
              class='w-36 h-36 object-cover mb-4 rounded-lg'
            />
            <h3 class='text-gray-700'>{{ 'selectedProduct.details' }}</h3>
          </ng-container>
          <ng-template >
          <h3 class='text-gray-500'>'Select a product to see the details'</h3>
          </ng-template>
        </div>
    </div>`
})
export class AppComponent {
  products: Product[] = [
    { id: 1, name: 'Product 1', image: 'https://via.placeholder.com/100', details: 'Details about Product 1' },
    { id: 2, name: 'Product 2', image: 'https://via.placeholder.com/100', details: 'Details about Product 2' },
    { id: 3, name: 'Product 3', image: 'https://via.placeholder.com/100', details: 'Details about Product 3' }
  ];



}
